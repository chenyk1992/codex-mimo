# codex-mimo 全场景测试计划 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use compose:subagent (recommended) or compose:execute to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 为 codex-mimo 项目的 166 个测试场景编写完整的单元测试，覆盖 8 个子系统，发现并记录问题。

**Architecture:** 使用 vitest + vi.mock() 模式，每个子系统一个测试文件。所有外部依赖（execa、fs、process）通过 mock 或 DI 注入。测试遵循现有项目的 mock 模式（vi.hoisted + vi.mock）。

**Tech Stack:** vitest, vi.mock, vi.hoisted, fs.mkdtempSync, execa (mocked)

## Global Constraints

- ESM-only: 所有 import 必须使用 `.js` 扩展名
- 测试文件放在 `test/unit/` 下，按子系统分目录
- 使用 `vi.hoisted()` + `vi.mock()` 模式 mock 外部依赖
- 临时目录使用 `fs.mkdtempSync(path.join(os.tmpdir(), "prefix-"))`
- 每个测试文件独立，可单独运行
- 发现的问题记录到 `docs/compose/specs/2026-06-24-test-plan-design.md` 的 Issue 表

```typescript
import { describe, expect, it, vi, beforeEach } from "vitest";

const mocks = vi.hoisted(() => ({
  execa: vi.fn()
}));

vi.mock("execa", () => ({ execa: mocks.execa }));

import { mimoHealthcheck } from "../../../src/codex/tools.js";

describe("mimo_healthcheck", () => {
  beforeEach(() => mocks.execa.mockReset());

  it("returns ok with version when mimo is available", async () => {
    mocks.execa.mockResolvedValue({ stdout: "mimo 1.2.3", stderr: "" });
    const result = await mimoHealthcheck({ cwd: "/project" });
    expect(result).toMatchObject({ ok: true, version: "mimo 1.2.3", cwd: "/project" });
  });

  it("returns ok=false when mimo is not installed", async () => {
    mocks.execa.mockRejectedValue(new Error("ENOENT"));
    const result = await mimoHealthcheck({ cwd: "/project" });
    expect(result).toMatchObject({ ok: false, error: "mimo not found or not working" });
  });

  it("uses process.cwd() when cwd is not specified", async () => {
    mocks.execa.mockResolvedValue({ stdout: "mimo 1.0.0", stderr: "" });
    const result = await mimoHealthcheck({});
    expect(result.cwd).toBe(process.cwd());
  });
});
```

```typescript
import fs from "node:fs";
import os from "node:os";
import path from "node:path";
import { describe, expect, it, vi, beforeEach, afterEach } from "vitest";

const mocks = vi.hoisted(() => ({
  execa: vi.fn(),
  runAndCapture: vi.fn()
}));

vi.mock("execa", () => ({ execa: mocks.execa }));
vi.mock("../../../../src/mimo/mimo-runner.js", () => ({
  runAndCapture: mocks.runAndCapture
}));

import { mimoReview } from "../../../../src/codex/tools.js";

const tempDirs: string[] = [];

describe("mimo_review", () => {
  beforeEach(() => {
    mocks.execa.mockReset();
    mocks.runAndCapture.mockReset();
  });

  afterEach(() => {
    for (const d of tempDirs.splice(0)) fs.rmSync(d, { recursive: true, force: true });
  });

  it("creates temp .diff file and passes as @attachment", async () => {
    const cwd = fs.mkdtempSync(path.join(os.tmpdir(), "review-"));
    tempDirs.push(cwd);
    const diff = "diff --git a/src/app.ts b/src/app.ts\n+const x = 1;\n";
    mocks.execa.mockResolvedValue({ exitCode: 0, stdout: diff, stderr: "" });
    mocks.runAndCapture.mockResolvedValue({
      sessionId: "ses_1", summary: "LGTM", changedFiles: [], commands: [], errors: [], exitCode: 0, raw: [{ type: "text" }]
    });

    const result = await mimoReview({ cwd, base: "HEAD" });
    expect(result.findings).toHaveLength(1);
    const runInput = mocks.runAndCapture.mock.calls[0][0];
    expect(runInput.files).toHaveLength(1);
    expect(runInput.message).toContain("attached");
  });

  it("throws when git diff fails", async () => {
    const cwd = fs.mkdtempSync(path.join(os.tmpdir(), "review-"));
    tempDirs.push(cwd);
    mocks.execa.mockResolvedValue({ exitCode: 128, stdout: "", stderr: "fatal: bad revision" });

    await expect(mimoReview({ cwd, base: "missing" })).rejects.toThrow("Git diff capture failed");
    expect(mocks.runAndCapture).not.toHaveBeenCalled();
  });

  it("throws when review produces no output", async () => {
    const cwd = fs.mkdtempSync(path.join(os.tmpdir(), "review-"));
    tempDirs.push(cwd);
    mocks.execa.mockResolvedValue({ exitCode: 0, stdout: "diff --git a/a b/a\n", stderr: "" });
    mocks.runAndCapture.mockResolvedValue({
      sessionId: null, summary: "Completed.", changedFiles: [], commands: [], errors: [], exitCode: 0, raw: []
    });

    await expect(mimoReview({ cwd, base: "HEAD" })).rejects.toThrow("produced no review output");
  });

  it("skips temp file when no diff content", async () => {
    const cwd = fs.mkdtempSync(path.join(os.tmpdir(), "review-"));
    tempDirs.push(cwd);
    mocks.execa.mockResolvedValue({ exitCode: 0, stdout: "No changes found.", stderr: "" });
    mocks.runAndCapture.mockResolvedValue({
      sessionId: null, summary: "No findings.", changedFiles: [], commands: [], errors: [], exitCode: 0, raw: [{ type: "text" }]
    });

    const result = await mimoReview({ cwd, base: "HEAD" });
    const runInput = mocks.runAndCapture.mock.calls[0][0];
    expect(runInput.files).toBeUndefined();
  });
});
```
